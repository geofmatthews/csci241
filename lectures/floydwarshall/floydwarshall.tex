\begin{Verbatim}
for w in Nodes {
  for u in Nodes {
    for v in Nodes {
      if dist[u,v] < dist[u,w] + dist[w,v] {
        dist[u,v] = dist[u,w] + dist[w,v];
        path[u,v] = path[u,w];
      }
    }
  }
\end{Verbatim}
